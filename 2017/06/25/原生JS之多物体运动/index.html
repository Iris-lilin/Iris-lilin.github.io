原生JS之多物体运动速度篇

发表于 2017-06-25 | 分类于 study

本节分别从速度动画和透明度动画两个方面讲多物体动画

速度初始代码

仿照简单动画写的多物体运动的代码如下

<script type="text/javascript">
window.onload=function(){
    var obj=document.getElementsByTagName("div");
    for(var i=0;i<obj.length;i++){
        obj[i].onmouseover=function(){
        move(this,400);
        }
        obj[i].onmouseout=function(){
        move(this,200);
        }
    }
}

    var timer=null;
    function move(obj,target){
        clearInterval(timer);
        timer=setInterval(function(){
            var speed=(target-obj.offsetWidth)/10;
            speed=speed<0?Math.floor(speed):Math.ceil(speed);
            if (obj.offsetLeft==target) {
                clearInterval(timer);
             }else{
                 obj.style.width=obj.offsetWidth+speed+"px";
             }
         },30)
    }

</script>

效果图


  测试时发现当鼠标移入到第一个div 时，他是正常运行的。但是如果现在又移动到第二个或者第三个div时候就会出现bug。

原因：
整个程序就一个定时器, 比如第一个div开始动了，第二个div 鼠标移入了 前一个定时器就被干掉了，那么自然就卡在那里了。
所以最大的问题就是整个程序就只有一个定时器。那么怎么解决这个问题呢？

  其实很简单，把定时器作为一个物体的属性加上,那么每个物体都有一个定时器在，当关闭定时器的时候是关闭物体上的定时器,开也是物体上的定时器

那么他们之间就可以完全互不干扰的运行。

速度改进代码

function move(obj,target){
    clearInterval(obj.timer);
    obj.timer=setInterval(function(){
        var speed=(target-obj.offsetWidth)/8;
        speed=speed<0?Math.floor(speed):Math.ceil(speed);
        if (obj.offsetLeft==target) {
            obj.clearInterval(timer);
         }else{
             obj.style.width=obj.offsetWidth+speed+"px";
         }
     },30)
}

这样就能达到我们期望的效果了，如图：

透明度初始代码

仿照简单物体透明度动画写的多物体运动的代码如下

<script type="text/javascript">
window.onload=function(){
    var obj=document.getElementsByTagName("div");
    for(var i=0;i<obj.length;i++){
        obj[i].onmouseover=function(){
        move(this,100);
        }
        obj[i].onmouseout=function(){
        move(this,30);
        }
    }
}

    var alpha=30;
    function move(obj,target){
        clearInterval(obj.timer);
        obj.timer=setInterval(function(){
            if (target>alpha) {
                speed=10;
            }else{
                speed=-10;
            }
            if (alpha==target) {
                obj.clearInterval(timer);
             }else{
                 alpha+=speed;
                 obj.style.filter="alpha(opacity"+alpha+")";
                 obj.style.opacity=alpha/100;
             }
         },30)
    }
 </script>

效果图


遇到的问题和速度动画一样，解决方法也是，所以直接贴代码

<script type="text/javascript">
window.onload=function(){
var obj=document.getElementsByTagName("div");
for(var i=0;i<obj.length;i++){
    obj[i].alpha=30;
    obj[i].onmouseover=function(){
    move(this,100);
    }
    obj[i].onmouseout=function(){
    move(this,30);
    }
}
}
function move(obj,target){
    clearInterval(obj.timer);
    obj.timer=setInterval(function(){
        if (target>obj.alpha) {
            speed=10;
        }else{
            speed=-10;
        }
        if (obj.alpha==target) {
            obj.clearInterval(timer);
         }else{
             obj.alpha+=speed;
             obj.style.filter="alpha(opacity"+obj.alpha+")";
             obj.style.opacity=obj.alpha/100;
         }
     },30)
}
</script>

效果图

经验小结

只要是多物体运动，所有的东西都不能公用

像timer一样定义一下，每一个obj都定义自己本身的alpha，就不会去争用一个属性了。
