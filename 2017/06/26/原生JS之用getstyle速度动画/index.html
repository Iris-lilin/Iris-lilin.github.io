原生JS之用getstyle速度动画

发表于 2017-06-26

  当元素只有宽高等基本属性时，之前的代码没有问题，然而一旦加了border、padding等属性之后，获取offsetHeight这些，可以说所有的offset这时候都不起作用了，这时候就需要用到getStyle这个已经封装好的函数。

getStyle代码

function getStyle(obj,attr){
    if (obj.currentStyle) {
        return obj.currentStyle[attr]; //IE
    }
    else{
        return getComputedStyle(obj,false)[attr]; //firefox、chrome
    }
}

改变样式代码

  window.onload=function(){
var obj=document.getElementsByTagName("div");
for(var i=0;i<obj.length;i++){
    //obj[i].offsetHeight=30;
    obj[i].onmouseover=function(){
    move(this,400);
    }
    obj[i].onmouseout=function(){
    move(this,200);
    }
}
}
var timer=null;
    function move(obj,target){

        clearInterval(obj.timer);
        obj.timer=setInterval(function(){
            var cur=parseInt(getStyle(obj,'width'));  //当前样式
            var speed=(target-cur)/10;
            speed=speed<0?Math.floor(speed):Math.ceil(speed);
            if (cur==target) {
                clearInterval(obj.timer);
             }else{
                 obj.style["width"]=cur+speed+"px"; //使用【】方便修改获取的属性
             }
         },30)
    }
