    position: absolute;

}

通过改变最左侧物体的left值实现动画

当鼠标移入的的时候left值加大，移出时减小。

    window.onload=function(){
    var obj=document.getElementById("div1");
    obj.onmouseover=function(){
        startMove();
    }
    obj.onmouseout=function(){
        endMove();
    }
}

具体实现代码如下

    function startMove(){
    clearInterval(timer);
    var obj=document.getElementById("div1");
    timer=setInterval(function(){
        if (obj.offsetLeft==0) {
            clearInterval(timer);
        }else{
            obj.style.left=obj.offsetLeft+5+"px";
        }
    },30)
}
function endMove(){
    clearInterval(timer);
    var obj=document.getElementById("div1");
        timer=setInterval(function(){
        if (obj.offsetLeft==-200) {
            clearInterval(timer);
        }else{
            obj.style.left=obj.offsetLeft-5+"px";
        }
    },30)
}

遇到的坑

  1. 每一次函数执行时都要先清除计时器，否则效果就是图片一直一闪一闪的改变left值
  2. 适当选择计时器的间隔时间，否则效果达不到期望值

精简版代码

在两个代码相像的情况下，可以把相同部分提取出来，不同部分作为参数传进去，JS代码如下：

<script type="text/javascript">
window.onload=function(){
    var obj=document.getElementById("div1");
    obj.onmouseover=function(){
        move(0,10);
    }
    obj.onmouseout=function(){
        move(-200,-10);
    }
}

    var timer=null;
    function move(target,speed){
        clearInterval(timer);
        var obj=document.getElementById("div1");
        timer=setInterval(function(){
        if (obj.offsetLeft==target) {
            clearInterval(timer);
        }else{
            obj.style.left=obj.offsetLeft+speed+"px";
            }
        },30)
    }

目前是传两个参数，而参数当然是越少越好，故再次精简如下：

var timer=null;
function move(target){
    clearInterval(timer);
    var obj=document.getElementById("div1");
    timer=setInterval(function(){
        var speed=0;
        if(obj.offsetLeft<target){
                speed=10;
        }else{
            speed=-10;
        }
        if (obj.offsetLeft==target) {
            clearInterval(timer);
         }else{
             obj.style.left=obj.offsetLeft+speed+"px";
         }
     },30)
}

##小结 ##

  1. 明确分析出物体属于何种运动类型；
  2. 实现匀速运动的关键点是采用定时器 setInnterval();
  3. 清空定时器 clearInterval() 的使用；
  4. 注意物体css需设置position，及left 、top，其是实现左、右、上、下运动必不可少的。
