原生JS之透明度动画

动画效果


     


元素原始透明度为30，当鼠标移入时利用定时器增加到100，移出时变为初始状态。

css样式代码

<style type="text/css">
*{
    margin: 0;
    padding: 0;
}
#div1{
    width: 200px;
    height: 200px;
    background-color: red;
    filter: alpha(opacity:30);  //IE浏览器满值为100
    opacity: 0.3;  //火狐、谷歌满值为1
}
</style>

JS代码

window.onload=function(){
    var obj=document.getElementById("div1");
    obj.onmouseover=function(){
    move(100);
    }
    obj.onmouseout=function(){
    move(30);
    }
}
var alpha=30;   //将当前透明度放在变量里和目标值比较
var timer=null;
function move(target){
    var obj=document.getElementById("div1");
    clearInterval(timer); //清除计时器，防止计时器叠加
    timer=setInterval(function(){
        var speed=0;
        if(alpha>target){
        speed= -10;
        }else{
            speed= 10;
        }
        if(alpha==target){
            clearInterval(timer);
        }
        else{
            alpha+=speed;
            obj.style.filter='alpha(opacity:'+alpha+')';
            obj.style.opacity=alpha/100;  
        }
    },30)
}

小结

  1. 浏览器兼容问题
  2. 没有的属性就可以封装在变量里，方便比较
